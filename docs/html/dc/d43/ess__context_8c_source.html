39 ess_error_t ess_context_create(ess_context_t* context, const char* name, const ess_format_t format) {
40  if(context == 0) { return ESS_ERROR_NULL; }
41 
42  ess_backend_facktory_t* backend = ess_backend_get_by_name(name);
43  if(backend == 0) {
44  ESP_LOGE(LOG_TAG,"Backend with name: %s not found", name);
45  return ESS_ERROR_NOBACKEND;
46  }
47 
48  if(backend->ess_backend_probe(format) == ESS_OK) {
49  if(backend->ess_backend_open(format) != ESS_OK) {
50  ESP_LOGE(LOG_TAG,"Backend with name: %s error open", name);
51  return ESS_ERROR_WRONG_FORMAT;
52  }
53  }
54  context->format = format;
55  context->status = ESS_CONTEXT_STATUS_RUN;
56  context->backend = backend;
57 
58  ESP_LOGI(LOG_TAG,"Backend with name: %s found and open", name);
59  return ESS_OK;
60 }
