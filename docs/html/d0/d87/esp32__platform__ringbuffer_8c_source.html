44 ess_error_t ess_platform_ringbuffer_create(ess_platform_ringbuffer_t* rng,
45  unsigned int length, ess_platform_ringbuffer_mode_t type, const char* name) {
46  if(rng == 0) return ESS_ERROR_NULL;
47  strncpy(rng->name, name, sizeof(rng->name) );
48 
49  rng->type = type;
50 
51  if(type == ESS_PLATFORM_RINGBUFFER_MODE_NOSPLIT)
52  rng->handle = xRingbufferCreate(length, RINGBUF_TYPE_NOSPLIT);
53  else if(type == ESS_PLATFORM_RINGBUFFER_MODE_ALLOWSPLIT)
54  rng->handle = xRingbufferCreate(length, RINGBUF_TYPE_ALLOWSPLIT);
55  else
56  rng->handle = xRingbufferCreate(length, RINGBUF_TYPE_BYTEBUF);
57 
58  rng->qhandle = xQueueCreateSet(3);
59  if (xRingbufferAddToQueueSetRead(rng->handle, rng->qhandle) != pdTRUE) {
60  printf("Failed to add to queue set\n");

62  }
63 
64  return rng->handle == 0 ? ESS_OK : ESS_ERROR;
65 }
