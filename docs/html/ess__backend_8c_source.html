53  ess_backend_facktory_t *factory;
54 
55  unsigned int i,n = 0;
56 
57  for( i = 0; i < ess_backend_get_size(); i++ ) {
58  factory = backends_list[i].getFactory();
59 
60  if(factory->ess_backend_probe(format) != ESS_BACKEND_OK) {
61  ESP_LOGE(LOG_TAG, "Failed to probe backend \"%s\"", backends_list[i].name);
62  continue;
63  } else {
64  ESP_LOGI(LOG_TAG,"Possible format for backend: \"%s\" ", backends_list[i].name);
65  if(backend) { backend[n] = factory; }
66  n++;
67  }
68 
69  }
70  ESP_LOGI(LOG_TAG,"%d/%d backends probe possible\n", n, ess_backend_get_size());
71 
72  if(n == 0) return ESS_BACKEND_ERROR_WRONG_FORMAT;
73  return n;
74 }
