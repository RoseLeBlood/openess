48  ess_socket_fam_t return_value = -1;
49  struct addrinfo hint, *result;
50  int af = -1;
51 
52  if ( hostname == NULL ) return -1;
53  memset(&hint,0,sizeof hint);
54  hint.ai_family = AF_UNSPEC;
55 
56  if ( 0 != (return_value = getaddrinfo(hostname,"0",&hint,&result))) return -1;
57  if ( result == NULL ) return -1;
58 
59  switch (result->ai_family) {
60  case AF_INET: af = ESS_SOCKET_FAMILY_IP4; break;
61  case AF_INET6: af= ESS_SOCKET_FAMILY_IP6; break;
62  };
63  return af;

76 ess_socket_error_t ess_socket_create_server(ess_socket_t* _socket) {
77  if(_socket == 0) return ESS_SOCKET_ERROR_NULL;
78  if(_socket->status != ESS_SOCKET_STATUS_CREATED || _socket->status == ESS_SOCKET_STATUS_STOPPED)
79  return ESS_SOCKET_ERROR_UNSPEC;
80 
81  struct addrinfo *result, *result_check, hints;
82  memset(&hints,0,sizeof(struct addrinfo));
83 
84  switch(_socket->protokol) {
85  case ESS_SOCKET_PROTO_STREAM: hints.ai_socktype = SOCK_STREAM; break;
86  case ESS_SOCKET_PROTO_DRAM: hints.ai_socktype = SOCK_DGRAM; break;
87  case ESS_SOCKET_PROTO_DRAM_LITE: hints.ai_socktype = SOCK_DGRAM; hints.ai_protocol = IPPROTO_UDPLITE; break;
88  default: _socket->status = ESS_SOCKET_STATUS_ERROR; return ESS_SOCKET_ERROR_UNSPEC_PROTOKOL;
89  };
90  switch ( _socket->family ) {
91  case ESS_SOCKET_FAMILY_IP4: hints.ai_family = AF_INET; break;
92  case ESS_SOCKET_FAMILY_IP6: hints.ai_family = AF_INET6; break;
93  case ESS_SOCKET_FAMILY_BOTH: hints.ai_family = AF_UNSPEC; break;
94  default: _socket->status = ESS_SOCKET_STATUS_ERROR; return ESS_SOCKET_ERROR_UNSPEC_FAMILY;
95  }
96  hints.ai_flags = AI_PASSIVE;
97 
98  char buffer[8];
99  snprintf(buffer, 8,"%d",_socket->port) ;
100  if ( (_socket->retval = getaddrinfo(_socket->hostname, buffer ,&hints,&result)) != 0 ) {
101  _socket->status = ESS_SOCKET_STATUS_ERROR;
102  return ESS_SOCKET_ERROR_GETADDR;
103  }
104 
105  for ( result_check = result; result_check != NULL; result_check = result_check->ai_next ) {
106  _socket->socket = socket(result_check->ai_family, result_check->ai_socktype, result_check->ai_protocol);
107  if ( _socket->socket < 0 ) continue;
108  if ( (_socket->retval = bind(_socket->socket, result_check->ai_addr,(socklen_t)result_check->ai_addrlen)) != 0 ) {
109  close(_socket->socket); continue;
110  }
111  if (_socket->protokol == ESS_SOCKET_PROTO_STREAM) {
112  if ( (_socket->retval = listen(_socket->socket, 128)) == 0) {
113  break;
114  } else {
115  close(_socket->socket);
116  }
117  }
118  }
119  if ( result_check == NULL ) {
120  freeaddrinfo(result);
121  _socket->status = ESS_SOCKET_STATUS_ERROR;
122  return ESS_SOCKET_ERROR_BIND;
123  }
124  freeaddrinfo(result);
125 
126  _socket->status = ESS_SOCKET_STATUS_LISTEN;
127  return ESS_SOCKET_ERROR_OK;
128 }

180 ess_socket_t* ess_socket_accept(ess_socket_t* server_socket, ess_socket_error_t* error_code, int flasg) {
181  if(server_socket == 0) { if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL; return 0; }
182 
183  ess_socket_t *client_socket;
184  int nbl = 0;
185  struct sockaddr_in incoming;
186  struct sockaddr_in6 incoming6;
187  unsigned int size_in6 = sizeof(struct sockaddr_in6);
188  unsigned int size_in = sizeof(struct sockaddr_in);
189 
190  //socklen_t addrlen = sizeof(struct sockaddr_storage);
191 
192  client_socket = (ess_socket_t*)malloc(sizeof(ess_socket_t));
193  if(client_socket == 0) return 0;
194 
195  client_socket->hostname_len = INET6_ADDRSTRLEN;
196 
197  if(server_socket->family == ESS_SOCKET_FAMILY_IP4) {
198 
199  if ( (client_socket->socket = accept(server_socket->socket, (struct sockaddr*)&incoming, (socklen_t *) &size_in)) != 0) {
200  if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL; return 0;
201  }
202 
203  if(server_socket->protokol == ESS_SOCKET_PROTO_STREAM) {
204  server_socket->port = ntohs( incoming.sin_port );
205  long addr = ntohl( incoming.sin_addr.s_addr );
206 
207  snprintf(client_socket->hostname, client_socket->hostname_len, "%03u.%03u.%03u.%03u",
208  (unsigned int) addr >> 24, (unsigned int) (addr >> 16) % 256, (unsigned int) (addr >> 8) % 256,
209  (unsigned int) addr % 256 );
210  client_socket->hostname_len = strlen(client_socket->hostname);
211  }
212 
213  } else {
214 
215  if ( (client_socket->socket = accept( server_socket->socket,(struct sockaddr *)&incoming6, (socklen_t *) &size_in6 )) != 0) {
216  if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL;
217  return 0;
218  }
219 
220  char addrbuf[INET6_ADDRSTRLEN];
221  addrbuf[0] = '\0';
222 
223  client_socket->port = ntohs( incoming6.sin6_port );
224 
225  if ( inet_ntop( AF_INET6, &(incoming6.sin6_addr), addrbuf, sizeof(addrbuf) )) {
226  ESP_LOGI("EssS", "client from :%s/%d", addrbuf, client_socket->port);
227  strncpy(client_socket->hostname, addrbuf, strlen(addrbuf) );
228  }
229  }
230 
231  if ( ioctl( client_socket->socket , FIONBIO, &nbl ) < 0 ) {
232  ESP_LOGE("EssS", "(%02d) couldn't turn on blocking for client", client_socket->socket );
233  client_socket->status = ESS_SOCKET_STATUS_ERROR;
234  if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_NULL; return 0; }
235 
236  close(client_socket->socket);
237  }
238  return client_socket;
239 }
