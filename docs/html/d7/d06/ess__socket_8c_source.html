65 ess_socket_error_t ess_socket_create_server( ess_socket_fam_t fam, ess_socket_pro_t protokoll,
66  const char* hostname, unsigned short port, ess_socket_t* _socket) {
67  if(_socket == 0) return ESS_SOCKET_ERROR_NULL;
68 
69  _socket->family = fam;
70  _socket->protokol = protokoll;
71  _socket->port = port;
72  _socket->status = ESS_SOCKET_STATUS_CREATED;
73  strncpy(_socket->hostname, hostname, strlen(hostname));
74  _socket->hostname_len = strlen(hostname);
75 
76  if(_socket->status != ESS_SOCKET_STATUS_CREATED || _socket->status == ESS_SOCKET_STATUS_STOPPED)
77  return ESS_SOCKET_ERROR_UNSPEC;
78 
79  struct addrinfo *result, *result_check, hints;
80  memset(&hints,0,sizeof(struct addrinfo));
81 
82  switch(_socket->protokol) {
83  case ESS_SOCKET_PROTO_STREAM: hints.ai_socktype = SOCK_STREAM; break;
84  case ESS_SOCKET_PROTO_DRAM: hints.ai_socktype = SOCK_DGRAM; break;
85  case ESS_SOCKET_PROTO_DRAM_LITE: hints.ai_socktype = SOCK_DGRAM; hints.ai_protocol = IPPROTO_UDPLITE; break;
86  default: _socket->status = ESS_SOCKET_STATUS_ERROR; return ESS_SOCKET_ERROR_UNSPEC_PROTOKOL;
87  };
88  switch ( _socket->family ) {
89  case ESS_SOCKET_FAMILY_IP4: hints.ai_family = AF_INET; break;
90  case ESS_SOCKET_FAMILY_IP6: hints.ai_family = AF_INET6; break;
91  case ESS_SOCKET_FAMILY_BOTH: hints.ai_family = AF_UNSPEC; break;
92  default: _socket->status = ESS_SOCKET_STATUS_ERROR; return ESS_SOCKET_ERROR_UNSPEC_FAMILY;
93  }
94  hints.ai_flags = AI_PASSIVE;
95 
96  char buffer[8]; buffer[0] = '\0'; buffer[0] = '\0';
97  snprintf(buffer, 8,"%d",_socket->port) ;
98  if ( (_socket->retval = getaddrinfo(_socket->hostname, buffer ,&hints,&result)) != 0 ) {
99  _socket->status = ESS_SOCKET_STATUS_ERROR;
100  return ESS_SOCKET_ERROR_GETADDR;
101  }
102 
103  for ( result_check = result; result_check != NULL; result_check = result_check->ai_next ) {
104  _socket->socket = socket(result_check->ai_family, result_check->ai_socktype, result_check->ai_protocol);
105  if ( _socket->socket < 0 ) continue;
106  if ( (_socket->retval = bind(_socket->socket, result_check->ai_addr,(socklen_t)result_check->ai_addrlen)) != 0 ) {
107  close(_socket->socket); continue;
108  }
109  if (_socket->protokol == ESS_SOCKET_PROTO_STREAM) {
110  if ( (_socket->retval = listen(_socket->socket, 128)) == 0) {
111  break;
112  } else {
113  close(_socket->socket);
114  }
115  }
116  }
117  if ( result_check == NULL ) {
118  freeaddrinfo(result);
119  _socket->status = ESS_SOCKET_STATUS_ERROR;
120  return ESS_SOCKET_ERROR_BIND;
121  }
122  freeaddrinfo(result);
123 
124  _socket->status = ESS_SOCKET_STATUS_LISTEN;
125  return ESS_SOCKET_ERROR_OK;
126 }

178 ess_socket_t* ess_socket_accept(ess_socket_t* server_socket, ess_socket_error_t* error_code) {
179  if(server_socket == 0) { if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL; return 0; }
180 
181  ess_socket_t *client_socket;
182  int nbl = 0;
183  struct sockaddr_in incoming;
184  struct sockaddr_in6 incoming6;
185  unsigned int size_in6 = sizeof(struct sockaddr_in6);
186  unsigned int size_in = sizeof(struct sockaddr_in);
187 
188  //socklen_t addrlen = sizeof(struct sockaddr_storage);
189 
190 
191  client_socket = (ess_socket_t*)malloc(sizeof(ess_socket_t));
192  if(client_socket == 0) return 0;
193 
194  client_socket->hostname_len = INET6_ADDRSTRLEN;
195 
196  if(server_socket->family == ESS_SOCKET_FAMILY_IP4) {
197 
198  if ( (client_socket->socket = accept(server_socket->socket, (struct sockaddr*)&incoming, (socklen_t *) &size_in)) != 0) {
199  if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL; return 0;
200  }
201 
202  if(server_socket->protokol == ESS_SOCKET_PROTO_STREAM) {
203  server_socket->port = ntohs( incoming.sin_port );
204  long addr = ntohl( incoming.sin_addr.s_addr );
205 
206  snprintf(client_socket->hostname, client_socket->hostname_len, "%03u.%03u.%03u.%03u",
207  (unsigned int) addr >> 24, (unsigned int) (addr >> 16) % 256, (unsigned int) (addr >> 8) % 256,
208  (unsigned int) addr % 256 );
209  client_socket->hostname_len = strlen(client_socket->hostname);
210  }
211 
212  } else {
213 
214  if ( (client_socket->socket = accept( server_socket->socket,(struct sockaddr *)&incoming6, (socklen_t *) &size_in6 )) != 0) {
215  if(error_code != 0) *error_code = ESS_SOCKET_ERROR_NULL;
216  return 0;
217  }
218 
219  char addrbuf[INET6_ADDRSTRLEN];
220  addrbuf[0] = '\0';
221 
222  client_socket->port = ntohs( incoming6.sin6_port );
223 
224  if ( inet_ntop( AF_INET6, &(incoming6.sin6_addr), addrbuf, sizeof(addrbuf) )) {
225  ESP_LOGI("EssS", "client from :%s/%d", addrbuf, client_socket->port);
226  strncpy(client_socket->hostname, addrbuf, strlen(addrbuf) );
227  }
228  }
229 
230  if ( ioctl( client_socket->socket , FIONBIO, &nbl ) < 0 ) {
231  ESP_LOGE("EssS", "(%02d) couldn't turn on blocking for client", client_socket->socket );
232  client_socket->status = ESS_SOCKET_STATUS_ERROR;
233  if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_NULL; return 0; }
234 
235  close(client_socket->socket);
236  }
237  return client_socket;
238 }

248  if(hostname == NULL) { if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_NULL; } return 0; }
249 
250  int sfd;
251  struct addrinfo hint, *result, *result_check;
252  ess_socket_t* new_socket;
253  char buffer[8]; buffer[0] = '\0';
254 
255  memset(&hint,0,sizeof hint);
256 
257  switch ( family ) {
258  case ESS_SOCKET_FAMILY_IP4: hint.ai_family = AF_INET; break;
259  case ESS_SOCKET_FAMILY_IP6: hint.ai_family = AF_INET6; break;
260  default: hint.ai_family = AF_UNSPEC;
261  }
262 
263 
264  hint.ai_socktype = SOCK_STREAM;
265 
266 
267  snprintf(buffer, 8,"%d",port) ;
268 
269  if ( getaddrinfo(hostname,buffer,&hint,&result) != 0) {
270  if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_GETADDR; }
271  return 0;
272  }
273 
274  for ( result_check = result; result_check != 0; result_check = result_check->ai_next ) { // go through the linked list of struct addrinfo elements
275  sfd = socket(result_check->ai_family, result_check->ai_socktype | flags, result_check->ai_protocol);
276 
277  if ( sfd < 0 ) continue;
278 
279  if ( connect(sfd,result_check->ai_addr,result_check->ai_addrlen) != -1) // connected without error
280  break;
281  close(sfd);
282  }
283  freeaddrinfo(result);
284 
285  if ( result_check == 0 ) {
286  if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_GETADDR; }
287  return 0;
288  }
289  // Yes :)
290  new_socket = (ess_socket_t*)malloc(sizeof(ess_socket_t));
291  if(new_socket == 0) {
292  if(error_code != 0) { *error_code = ESS_SOCKET_ERROR_OUTOF_MEM; }
293  return 0;
294  }
295  new_socket->socket = sfd;
296 
297  strncpy(new_socket->hostname, hostname, strlen(hostname));
298 
299  new_socket->port = port;

308 
309  if (fam != ESS_SOCKET_FAMILY_IP6 && fam != ESS_SOCKET_FAMILY_IP4) {
