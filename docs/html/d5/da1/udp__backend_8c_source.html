43 ess_backend_error_t ess_backend_udp_probe(const ess_format_t format) {
44  switch (format) {
45  case ESS_FORMAT_MONO_96000_8:
46  case ESS_FORMAT_MONO_96000_16:
47  case ESS_FORMAT_MONO_96000_24:
48  case ESS_FORMAT_STEREO_96000_8:
49  case ESS_FORMAT_STEREO_96000_16:

51  break;
52  default:
53  return ESS_BACKEND_OK;

55  printf("Format not support for backend: udp (%s)\n", ess_format_to_string(format));

58 ess_backend_error_t ess_backend_udp_open(const ess_format_t format) {

61 #if ESS_BACKEND_UDP_FAMILY == ESS_FAMILY_BOTH

67 #else
68  error = ess_socket_create_server( ESS_SOCKET_FAMILY_IP6, ESS_SOCKET_PROTO_DRAM, "::",
69  ESS_BACKEND_UDP_PORT, &(((udp_backend_t*)_udp_backend_config.user_daten)->socket));
70 #endif

72 }
