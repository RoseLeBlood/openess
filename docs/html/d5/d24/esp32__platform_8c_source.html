57 int IRAM_ATTR ess_platform_nsleep(const struct timespec *req, struct timespec *rem) {
58  struct timeval start, end;
59 
60  if ((req->tv_nsec < 0) || (req->tv_nsec > 999999999)) {
61  errno = EINVAL;
62 
63  return -1;
64  }
65 
66  // Get time in msecs
67  uint32_t msecs;
68 
69  msecs = req->tv_sec * 1000;
70  msecs += (req->tv_nsec + 999999) / 1000000;
71 
72  if (rem != NULL) {
73  gettimeofday(&start, NULL);
74  }
75 
76  vTaskDelay(msecs / portTICK_PERIOD_MS);
77  if (rem != NULL) {
78  rem->tv_sec = 0;
79  rem->tv_nsec = 0;
80 
81  gettimeofday(&end, NULL);
82 
83  uint32_t elapsed = (end.tv_sec - start.tv_sec) * 1000
84  + ((end.tv_usec - start.tv_usec) / 1000000);
85 
86  if (elapsed < msecs) {
87  if (elapsed > 1000) {
88  rem->tv_sec = elapsed / 1000;
89  elapsed -= rem->tv_sec * 1000;
90  rem->tv_nsec = elapsed * 1000000;
91  }
92  }
93  }
94  errno = EINTR;
95  return ESS_OK;
96 }
