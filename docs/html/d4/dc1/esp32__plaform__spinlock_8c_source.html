57 ess_error_t ess_platform_spinlock_lock(ess_platform_spinlock_t* spi) {
58  if(spi == 0) return ESS_ERROR_NULL;
59  if(spi->handle == 0) return ESS_ERROR_NULL;
60 
61  if (xPortInIsrContext()) {
62  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
63  xSemaphoreTakeFromISR( spi->handle, &xHigherPriorityTaskWoken );
64  if(xHigherPriorityTaskWoken)
65  _frxt_setup_switch();
66  } else {
67  xSemaphoreTake(spi->handle, portMAX_DELAY);
68  }
69 
70  return ESS_OK;
71 }
72 ess_error_t ess_platform_spinlock_unlock(ess_platform_spinlock_t* spi) {
