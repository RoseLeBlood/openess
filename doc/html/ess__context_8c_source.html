59 ess_context_error_t ess_context_init_ex(ess_context_t* context, ess_backend_facktory_t* backend) {
60  if(backend == 0 && backend == 0) {
61  context->last_error = ESS_CONTEXT_ERRORNOBACKEND;
62  return ESS_CONTEXT_ERRORNOBACKEND;
63  }
64 
65  if(backend->ess_backend_open(context->format) != 0) {
66  ESP_LOGE(LOG_TAG,"Error to open backend");
67  return -1;
68  } else {
69  context->backend = backend;
70  context->status = ESS_CONTEXT_STATUS_RUN;
71  ESP_LOGI(LOG_TAG,"Possible to open backend");
72  }
73  context->last_error = ESS_CONTEXT_ERROR_OK;
74  return ESS_CONTEXT_ERROR_OK;
